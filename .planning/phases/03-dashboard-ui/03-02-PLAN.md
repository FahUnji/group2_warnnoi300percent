useEffect(() => {
  if (!projects.length) return;
  const initial = {};
  projects.forEach(p => { initial[p.key] = { open: 0, critical: 0, loading: true }; });
  setBugStats(initial);

  projects.forEach(project => {
    fetch(`/api/bugs/${project.key}`)
      .then(r => r.json())
      .then(data => {
        if (data.ok) {
          const bugs = data.bugs;
          const open = bugs.filter(b =>
            ['open', 'to do'].includes((b.status || '').toLowerCase())
          ).length;
          const critical = bugs.filter(b =>
            ['critical', 'highest'].includes((b.priority || '').toLowerCase())
          ).length;
          setBugStats(prev => ({
            ...prev,
            [project.key]: { open, critical, loading: false }
          }));
        } else {
          setBugStats(prev => ({
            ...prev,
            [project.key]: { open: 0, critical: 0, loading: false }
          }));
        }
      })
      .catch(() => {
        setBugStats(prev => ({
          ...prev,
          [project.key]: { open: 0, critical: 0, loading: false }
        }));
      });
  });
}, [projects]);

extractProjectKey + handleConnect (from 03-PATTERNS.md "handleConnect pattern"):
```jsx
function extractProjectKey(input) {
  const trimmed = input.trim();
  if (trimmed.includes('/')) {
    const parts = trimmed.split('/').filter(Boolean);
    return parts[parts.length - 1].toUpperCase();
  }
  return trimmed.toUpperCase();
}

async function handleConnect() {
  if (!connectInput.trim() || connecting) return;
  const key = extractProjectKey(connectInput);
  setConnecting(true);
  setConnectStatus({ type: 'loading', message: 'Connecting…' });
  try {
    const resp = await fetch(`/api/sync/${key}`, { method: 'POST' });
    const data = await resp.json();
    if (data.ok) {
      setConnectStatus({ type: 'success', message: `Project ${key} synced successfully.` });
      setConnectInput('');
      fetch('/api/projects')
        .then(r => r.json())
        .then(d => { if (d.ok) setProjects(d.projects); })
        .catch(() => {});
    } else {
      setConnectStatus({ type: 'error', message: `${data.detail || 'Sync failed'}. Check project key and try again.` });
    }
  } catch {
    setConnectStatus({ type: 'error', message: 'Connection failed. Check project key and try again.' });
  } finally {
    setConnecting(false);
  }
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| connectInput (user text) → URL path segment | Raw user input flows into fetch(`/api/sync/${key}`) URL construction |
| API response data.detail → DOM text | Error message from backend is rendered directly in status line |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-06 | Tampering | extractProjectKey() | mitigate | Input is .trim().toUpperCase() before use in fetch URL. Backend is authoritative validator; frontend sanitization is defense-in-depth. The extracted key is used only as a URL path segment — no SQL, no eval, no innerHTML. |
| T-03-07 | Information Disclosure | data.detail in error message | accept | data.detail comes from the same-origin backend over a trusted internal API. Error text is displayed in a status line visible only to the authenticated user who triggered the action. No secret leakage risk. |
| T-03-08 | Denial of Service | projects.forEach parallel fetches | mitigate | If GET /api/projects returns many projects, parallel fetch calls could saturate connections. Mitigation: the parallel forEach loop is per-project fan-out; browser enforces max 6 concurrent connections per origin. For Phase 3 the project count is bounded (QA tool, not public). Accept residual risk; Phase 4 can add batching if needed. |
| T-03-09 | Spoofing | fetch response data.ok check | accept | Response is from same-origin backend; no CSRF risk for GET requests. POST /api/sync uses same-origin fetch with no custom headers — browser SOP applies. Backend should validate session (already implemented in Phase 1). |
</threat_model>
